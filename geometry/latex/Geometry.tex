\subsection{Geo geo}
\begin{enumerate}

\item
Ley de senos
\begin{equation}
  \frac{a}{sinA} = \frac{b}{sinB} = \frac{c}{sinC}
\end{equation}

\item
Ley de cosenos
\begin{equation}
  a^2 =  b^2 + c^2 - 2bc cos(A)
\end{equation}

Teorema de Pick: (Area, puntos interiores y puntos en el borde)
$A=I+\frac{B}{2}-1$

\item
Calcular puntos enteros sobre una recta A(x, y) -> B(x, y):
gcd(A.x - B.x, A.y - B.y)

\end{enumerate}
